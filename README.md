# VERSION

version 0.193291

# SYNOPSIS

    my $anthill = Anthill->new('sqlite-anthill.db');
        $anthill
        ->ant( 'ant#1'=> [ foo, {bar => 'baz'} ] )
        ->start_args(perl_command => {
                        worker => q{-E"say 'hello from ant!'"}
                })
        ->start
        ->wait;
